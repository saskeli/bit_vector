101  bool at(const uint32_t i) const {
102  if constexpr (buffer_size != 0) {
103  uint64_t index = i;
104  for (uint8_t idx = 0; idx < buffer_count_; idx++) {
105  uint64_t b = buffer_index(buffer_[idx]);
106  if (b == i) {
107  if (buffer_is_insertion(buffer_[idx])) {
108  return buffer_value(buffer_[idx]);
109  }
110  index++;
111  } else if (b < i) {
112  [[likely]] index -=
113  buffer_is_insertion(buffer_[idx]) * 2 - 1;
114  } else {
115  break;
116  }
117  }
118  return MASK & (data_[index / WORD_BITS] >> (index % WORD_BITS));
119  }

142  void insert(const uint64_t i, const bool x) {
143 #ifdef DEBUG
144  if (size_ >= capacity_ * WORD_BITS) {
145  std::cerr << "Overflow. Reallocate before adding elements"
146  << "\n";
147  std::cerr << "Attempted to insert to " << i << std::endl;
148  assert(size_ < capacity_ * WORD_BITS);
149  }
150 #endif
151  if (i == size_) {
152  // Convert to append if applicable.
153  push_back(x);
154  [[unlikely]] return;
155  }
156  p_sum_ += x ? 1 : 0;
157  if constexpr (buffer_size != 0) {
158  uint8_t idx = buffer_count_;
159  while (idx > 0) {
160  uint64_t b = buffer_index(buffer_[idx - 1]);
161  if (b > i ||
162  (b == i && buffer_is_insertion(buffer_[idx - 1]))) {
163  [[likely]] set_buffer_index(b + 1, idx - 1);
164  } else {
165  break;
166  }
167  idx--;
168  }
169  size_++;
170  if (idx == buffer_count_) {
171  buffer_[buffer_count_] = create_buffer(i, 1, x);
172  [[likely]] buffer_count_++;
173  } else {
174  insert_buffer(idx, create_buffer(i, 1, x));
175  }
176  if (buffer_count_ >= buffer_size) [[unlikely]]
177  commit();
178  } else {
179  // If there is no buffer, a simple linear time insertion is done
180  // instead.
181  size_++;
182  auto target_word = i / WORD_BITS;
183  auto target_offset = i % WORD_BITS;
184  for (size_t j = capacity_ - 1; j > target_word; j--) {
185  data_[j] <<= 1;
186  data_[j] |= (data_[j - 1] >> 63);
187  }
188  data_[target_word] =
189  (data_[target_word] & ((MASK << target_offset) - 1)) |
190  ((data_[target_word] & ~((MASK << target_offset) - 1)) << 1);
191  data_[target_word] |= x ? (MASK << target_offset) : uint64_t(0);
192  }
193  }

209  bool remove(const uint64_t i) {
210  if constexpr (buffer_size != 0) {
211  bool x = this->at(i);
212  p_sum_ -= x;
213  --size_;
214  uint8_t idx = buffer_count_;
215  while (idx > 0) {
216  uint64_t b = buffer_index(buffer_[idx - 1]);
217  if (b == i) {
218  if (buffer_is_insertion(buffer_[idx - 1])) {
219  delete_buffer_element(idx - 1);
220  return x;
221  } else {
222  [[likely]] break;
223  }
224  } else if (b < i) {
225  break;
226  } else {
227  [[likely]] set_buffer_index(b - 1, idx - 1);
228  }
229  idx--;
230  }
231  if (idx == buffer_count_) {
232  buffer_[idx] = create_buffer(i, 0, x);
233  buffer_count_++;
234  } else {
235  [[likely]] insert_buffer(idx, create_buffer(i, 0, x));
236  }
237  if (buffer_count_ >= buffer_size) [[unlikely]]
238  commit();
239  return x;
240  } else {
241  // If buffer does not exits. A simple linear time removal is done
242  // instead.
243  uint64_t target_word = i / WORD_BITS;
244  uint64_t target_offset = i % WORD_BITS;
245  bool x = MASK & (data_[target_word] >> target_offset);
246  p_sum_ -= x;
247  data_[target_word] =
248  (data_[target_word] & ((MASK << target_offset) - 1)) |
249  ((data_[target_word] >> 1) & (~((MASK << target_offset) - 1)));
250  data_[target_word] |= (uint32_t(capacity_ - 1) > target_word)
251  ? (data_[target_word + 1] << 63)
252  : 0;
253  for (size_t j = target_word + 1; j < uint32_t(capacity_ - 1); j++) {
254  data_[j] >>= 1;
255  data_[j] |= data_[j + 1] << 63;
256  }
257  data_[capacity_ - 1] >>=
258  (uint64_t(capacity_ - 1) > target_word) ? 1 : 0;
259  size_--;
260  return x;
261  }

276  int set(const uint64_t i, const bool x) {
277  uint64_t idx = i;
278  if constexpr (buffer_size != 0) {
279  // If buffer exists, the index needs for the underlying structure
280  // needs to be modified. And if there exists an insertion to this
281  // location in the buffer, the insertion can simply be amended.
282  for (uint8_t j = 0; j < buffer_count_; j++) {
283  uint64_t b = buffer_index(buffer_[j]);
284  if (b < i) {
285  [[likely]] idx += buffer_is_insertion(buffer_[j]) ? -1 : 1;
286  } else if (b == i) {

288  if (buffer_value(buffer_[j]) != x) {
289  int change = x ? 1 : -1;

292  [[likely]] return change;

325  uint64_t rank(const uint64_t n) const {
326  uint64_t count = 0;
327 
328  uint64_t idx = n;
329  if constexpr (buffer_size != 0) {
330  for (uint8_t i = 0; i < buffer_count_; i++) {
331  if (buffer_index(buffer_[i]) >= n) {
332  [[unlikely]] break;
333  }

335  // base on buffer contents.
336  if (buffer_is_insertion(buffer_[i])) {
337  idx--;

339  } else {

341  count -= buffer_value(buffer_[i]);

346  uint64_t target_offset = idx % WORD_BITS;
347  if constexpr (avx) {

373  uint32_t select(const uint32_t x) const {
374  uint32_t pop = 0;
375  uint32_t pos = 0;
376  uint8_t current_buffer = 0;
377  int8_t a_pos_offset = 0;
378 
379  // Step one 64-bit word at a time considering the buffer until pop >= x
380  for (uint64_t j = 0; j < capacity_; j++) {
381  pop += __builtin_popcountll(data_[j]);
382  pos += 64;
383  if constexpr (buffer_size != 0) {
384  for (uint8_t b = current_buffer; b < buffer_count_; b++) {
385  uint64_t b_index = buffer_index(buffer_[b]);
386  if (b_index < pos) {
387  if (buffer_is_insertion(buffer_[b])) {
388  pop += buffer_value(buffer_[b]);
389  pos++;
390  a_pos_offset--;
391  } else {
392  pop -=
393  (data_[(b_index + a_pos_offset) / WORD_BITS] >>
394  ((b_index + a_pos_offset) % WORD_BITS)) &
395  MASK;
396  pos--;
397  a_pos_offset++;
398  }
399  [[unlikely]] current_buffer++;
400  } else {
401  break;
402  }
403  }
404  }
405  if (pop >= x) [[unlikely]]
406  break;
407  }
408 
409  // Make sure we have not overshot the logical end of the structure.
410  pos = size_ < pos ? size_ : pos;
411 
412  // Decrement one bit at a time until we can't anymore without going
413  // under x.
414  pos--;
415  while (pop >= x && pos < capacity_ * 64) {
416  pop -= at(pos);
417  pos--;
418  }
419  return ++pos;
420  }

491  void clear_first(uint64_t elems) {
492  // Important to ensure that all trailing data gets zeroed out to not
493  // cause undefined behaviour for select or copy operations.
494  uint64_t ones = rank(elems);
495  uint64_t words = elems / WORD_BITS;
496 
497  if (elems % WORD_BITS == 0) {
498  // If the data to remove happens to align with 64-bit words the
499  // removal can be done by simply shuffling elements. This could be
500  // memmove instead but testing indicates that it's not actually
501  // faster in this case.
502  for (uint64_t i = 0; i < capacity_ - words; i++) {
503  data_[i] = data_[i + words];
504  }
505  for (uint64_t i = capacity_ - words; i < capacity_; i++) {
506  data_[i] = 0;
507  }
508  } else {
509  // clear elem bits at start of data
510  for (uint64_t i = 0; i < words; i++) {
511  data_[i] = 0;
512  }
513  uint64_t tail = elems % WORD_BITS;
514  uint64_t tail_mask = (MASK << tail) - 1;
515  data_[words] &= ~tail_mask;
516 
517  // Copy data backwars by elem bits
518  uint64_t words_to_shuffle = capacity_ - words - 1;
519  for (uint64_t i = 0; i < words_to_shuffle; i++) {
520  data_[i] = data_[words + i] >> tail;
521  data_[i] |= (data_[words + i + 1]) << (WORD_BITS - tail);
522  }
523  data_[capacity_ - words - 1] = data_[capacity_ - 1] >> tail;
524  for (uint64_t i = capacity_ - words; i < capacity_; i++) {
525  data_[i] = 0;
526  }
527  [[unlikely]] (void(0));
528  }
529  size_ -= elems;
530  p_sum_ -= ones;
531  }

546  void transfer_append(leaf* other, uint64_t elems) {
547  commit();
548  other->commit();
549 
550  uint64_t* o_data = other->data();
551  uint64_t split_point = size_ % WORD_BITS;
552  uint64_t target_word = size_ / WORD_BITS;
553  uint64_t copy_words = elems / WORD_BITS;
554  uint64_t overhang = elems % WORD_BITS;
555  // Branching depending on word alignment of source and target.
556  if (split_point == 0) {
557  // Copy target is word aligned
558  for (size_t i = 0; i < copy_words; i++) {
559  data_[target_word++] = o_data[i];
560  p_sum_ += __builtin_popcountll(o_data[i]);
561  }
562  if (overhang != 0) {
563  // Copy source is not word aligned
564  data_[target_word] =
565  o_data[copy_words] & ((MASK << overhang) - 1);
566  [[unlikely]] p_sum_ += __builtin_popcountll(data_[target_word]);
567  }
568  } else {
569  // Copy target is not word aligned
570  for (size_t i = 0; i < copy_words; i++) {
571  data_[target_word++] |= o_data[i] << split_point;
572  data_[target_word] |= o_data[i] >> (WORD_BITS - split_point);
573  p_sum_ += __builtin_popcountll(o_data[i]);
574  }
575  if (elems % WORD_BITS != 0) {
576  // Copy source is not word aligned.
577  uint64_t to_write =
578  o_data[copy_words] & ((MASK << overhang) - 1);
579  p_sum_ += __builtin_popcountll(to_write);
580  data_[target_word++] |= to_write << split_point;
581  if (target_word < capacity_) {
582  data_[target_word] |= to_write >> (WORD_BITS - split_point);
583  }
584  [[likely]] ((void)0);
585  }
586  }
587  size_ += elems;
588  other->clear_first(elems);
589  }

629  void transfer_prepend(leaf* other, uint64_t elems) {
630  commit();
631  other->commit();
632  uint64_t* o_data = other->data();
633  uint64_t words = elems / 64;
634  // Make space for new data
635  for (uint64_t i = capacity_ - 1; i >= words; i--) {
636  data_[i] = data_[i - words];
637  if (i == 0) break;
638  }
639  for (uint64_t i = 0; i < words; i++) {
640  data_[i] = 0;
641  }
642  uint64_t overflow = elems % 64;
643  if (overflow > 0) {
644  for (uint64_t i = capacity_ - 1; i > words; i--) {
645  data_[i] <<= overflow;
646  data_[i] |= data_[i - 1] >> (64 - overflow);
647  }
648  data_[words] <<= overflow;
649  [[likely]] (void(0));
650  }
651  // Copy over data from sibling
652  uint64_t source_word = other->size();
653  uint64_t source_offset = source_word % 64;
654  source_word /= 64;
655  // Somewhat complicated case handling based on target and source word
656  // alignment. Additional complications compared to transfer append is
657  // due to the both the start point and end point for the copy source not
658  // being word aligned.
659  if (source_offset == 0) {
660  // Source is word aligned.
661  if (overflow == 0) {
662  // Target is word aligned.
663  for (uint64_t i = words - 1; i < words; i--) {
664  data_[i] = o_data[--source_word];
665  p_sum_ += __builtin_popcountll(data_[i]);
666  }
667  } else {
668  // Target is not word aligned.
669  source_word--;
670  for (uint64_t i = words; i > 0; i--) {
671  p_sum_ += __builtin_popcountll(o_data[source_word]);
672  data_[i] |= o_data[source_word] >> (64 - overflow);
673  data_[i - 1] |= o_data[source_word--] << overflow;
674  }
675  uint64_t w = o_data[source_word] >> (64 - overflow);
676  p_sum_ += __builtin_popcountll(w);
677  [[likely]] data_[0] |= w;
678  }
679  [[unlikely]] (void(0));
680  } else {
681  // Source is not word aligned
682  if (overflow == 0) {
683  // Target is word aligned
684  for (uint64_t i = words - 1; i < words; i--) {
685  data_[i] = o_data[source_word] << (64 - source_offset);
686  data_[i] |= o_data[--source_word] >> source_offset;
687  p_sum_ += __builtin_popcountll(data_[i]);
688  }
689  } else {
690  // Target is not word aligned
691  uint64_t w;
692  for (uint64_t i = words; i > 0; i--) {
693  w = o_data[source_word] << (64 - source_offset);
694  w |= o_data[--source_word] >> source_offset;
695  p_sum_ += __builtin_popcountll(w);
696  data_[i] |= w >> (64 - overflow);
697  data_[i - 1] |= w << overflow;
698  }
699  w = o_data[source_word] << (64 - source_offset);
700  w |= o_data[--source_word] >> source_offset;
701  w >>= 64 - overflow;
702  p_sum_ += __builtin_popcountll(w);
703  [[likely]] data_[0] |= w;
704  }
705  }
706  size_ += elems;
707  other->clear_last(elems);
708  }

763  void commit() {
764  // Complicated bit manipulation but whacha gonna do. Hopefully won't
765  // need to debug this anymore.
766  if constexpr (buffer_size == 0) return;
767  if (buffer_count_ == 0) [[unlikely]]
768  return;
769 
770  uint64_t overflow = 0;
771  uint8_t overflow_length = 0;
772  uint8_t underflow_length = 0;
773  uint8_t current_index = 0;
774  uint32_t buf = buffer_[current_index];
775  uint64_t target_word = buffer_index(buf) / WORD_BITS;
776  uint64_t target_offset = buffer_index(buf) % WORD_BITS;
777 
778  uint64_t words = size_ / WORD_BITS;
779  words += size_ % WORD_BITS > 0 ? 1 : 0;
780  for (uint64_t current_word = 0; current_word < words; current_word++) {
781  uint64_t underflow =
782  current_word + 1 < capacity_ ? data_[current_word + 1] : 0;
783  if (overflow_length) {
784  [[likely]] underflow =
785  (underflow << overflow_length) |
786  (data_[current_word] >> (64 - overflow_length));
787  }
788 
789  uint64_t new_overflow = 0;
790  // If buffers need to be commit to this word:
791  if (current_word == target_word && current_index < buffer_count_) {
792  uint64_t word =
793  underflow_length
794  ? (data_[current_word] >> underflow_length) |
795  (underflow << (64 - underflow_length))
796  : (data_[current_word] << overflow_length) | overflow;
797  underflow >>= underflow_length;
798  uint64_t new_word = 0;
799  uint8_t start_offset = 0;
800  // While there are buffers for this word
801  while (current_word == target_word) {
802  new_word |=
803  (word << start_offset) & ((MASK << target_offset) - 1);
804  word = (word >> (target_offset - start_offset)) |
805  (target_offset == 0 ? 0
806  : target_offset - start_offset == 0
807  ? 0
808  : (underflow
809  << (64 - (target_offset - start_offset))));
810  underflow >>= target_offset - start_offset;
811  if (buffer_is_insertion(buf)) {
812  if (buffer_value(buf)) {
813  new_word |= MASK << target_offset;
814  }
815  start_offset = target_offset + 1;
816  if (underflow_length) [[unlikely]]
817  underflow_length--;
818  else
819  overflow_length++;
820  } else {
821  word >>= 1;
822  word |= underflow << 63;
823  underflow >>= 1;
824  if (overflow_length)
825  overflow_length--;
826  else [[likely]]
827  underflow_length++;
828  start_offset = target_offset;
829  }
830  current_index++;
831  if (current_index >= buffer_count_) [[unlikely]]
832  break;
833  buf = buffer_[current_index];
834  target_word = buffer_index(buf) / WORD_BITS;
835  [[unlikely]] target_offset = buffer_index(buf) % WORD_BITS;
836  }
837  new_word |=
838  start_offset < 64 ? (word << start_offset) : uint64_t(0);
839  new_overflow = overflow_length ? data_[current_word] >>
840  (64 - overflow_length)
841  : 0;
842  [[unlikely]] data_[current_word] = new_word;
843  } else {
844  if (underflow_length) {
845  data_[current_word] =
846  (data_[current_word] >> underflow_length) |
847  (underflow << (64 - underflow_length));
848  } else if (overflow_length) {
849  new_overflow =
850  data_[current_word] >> (64 - overflow_length);
851  [[likely]] data_[current_word] =
852  (data_[current_word] << overflow_length) | overflow;
853  } else {
854  overflow = 0;
855  }
856  }
857  overflow = new_overflow;
858  }
859  if (capacity_ > words) [[likely]]
860  data_[words] = 0;
861  buffer_count_ = 0;
862  }
