98  bool at(const uint32_t i) const {
99  if constexpr (buffer_size != 0) {
100  uint64_t index = i;
101  for (uint8_t idx = 0; idx < buffer_count_; idx++) {
102  uint64_t b = buffer_index(buffer_[idx]);
103  if (b == i) {
104  if (buffer_is_insertion(buffer_[idx])) {
105  return buffer_value(buffer_[idx]);
106  }
107  index++;
108  } else if (b < i) {
109  [[likely]] index -=
110  buffer_is_insertion(buffer_[idx]) * 2 - 1;
111  } else {
112  break;
113  }
114  }
115  return MASK & (data_[index / WORD_BITS] >> (index % WORD_BITS));
116  }
117  return MASK & (data_[i / WORD_BITS] >> (i % WORD_BITS));
118  }

139  void insert(const uint64_t i, const bool x) {
140 #ifdef DEBUG
141  if (size_ >= capacity_ * WORD_BITS) {
142  std::cerr << "Overflow. Reallocate before adding elements"
143  << "\n";
144  std::cerr << "Attempted to insert to " << i << std::endl;
145  assert(size_ < capacity_ * WORD_BITS);
146  }
147 #endif
148  if (i == size_) {
149  // Convert to append if applicable.
150  push_back(x);
151  [[unlikely]] return;
152  }
153  p_sum_ += x ? 1 : 0;
154  if constexpr (buffer_size != 0) {
155  uint8_t idx = buffer_count_;
156  while (idx > 0) {
157  uint64_t b = buffer_index(buffer_[idx - 1]);
158  if (b > i ||
159  (b == i && buffer_is_insertion(buffer_[idx - 1]))) {
160  [[likely]] set_buffer_index(b + 1, idx - 1);
161  } else {
162  break;
163  }
164  idx--;
165  }
166  size_++;
167  if (idx == buffer_count_) {
168  buffer_[buffer_count_] = create_buffer(i, 1, x);
169  [[likely]] buffer_count_++;
170  } else {
171  insert_buffer(idx, create_buffer(i, 1, x));
172  }
173  if (buffer_count_ >= buffer_size) [[unlikely]]
174  commit();
175  } else {
176  // If there is no buffer, a simple linear time insertion is done
177  // instead.
178  size_++;
179  auto target_word = i / WORD_BITS;
180  auto target_offset = i % WORD_BITS;
181  for (size_t j = capacity_ - 1; j > target_word; j--) {
182  data_[j] <<= 1;
183  data_[j] |= (data_[j - 1] >> 63);
184  }
185  data_[target_word] =
186  (data_[target_word] & ((MASK << target_offset) - 1)) |
187  ((data_[target_word] & ~((MASK << target_offset) - 1)) << 1);
188  data_[target_word] |= x ? (MASK << target_offset) : uint64_t(0);
189  }

206  bool remove(const uint64_t i) {
207  if constexpr (buffer_size != 0) {
208  bool x = this->at(i);
209  p_sum_ -= x;
210  --size_;
211  uint8_t idx = buffer_count_;
212  while (idx > 0) {
213  uint64_t b = buffer_index(buffer_[idx - 1]);
214  if (b == i) {
215  if (buffer_is_insertion(buffer_[idx - 1])) {
216  delete_buffer_element(idx - 1);
217  return x;
218  } else {
219  [[likely]] break;
220  }
221  } else if (b < i) {
222  break;
223  } else {
224  [[likely]] set_buffer_index(b - 1, idx - 1);
225  }
226  idx--;
227  }
228  if (idx == buffer_count_) {
229  buffer_[idx] = create_buffer(i, 0, x);
230  buffer_count_++;
231  } else {
232  [[likely]] insert_buffer(idx, create_buffer(i, 0, x));
233  }
234  if (buffer_count_ >= buffer_size) [[unlikely]]
235  commit();
236  return x;
237  } else {
238  // If buffer does not exits. A simple linear time removal is done
239  // instead.
240  uint64_t target_word = i / WORD_BITS;
241  uint64_t target_offset = i % WORD_BITS;
242  bool x = MASK & (data_[target_word] >> target_offset);
243  p_sum_ -= x;
244  data_[target_word] =
245  (data_[target_word] & ((MASK << target_offset) - 1)) |
246  ((data_[target_word] >> 1) & (~((MASK << target_offset) - 1)));
247  data_[target_word] |= (uint32_t(capacity_ - 1) > target_word)
248  ? (data_[target_word + 1] << 63)
249  : 0;
250  for (size_t j = target_word + 1; j < uint32_t(capacity_ - 1); j++) {
251  data_[j] >>= 1;
252  data_[j] |= data_[j + 1] << 63;
253  }
254  data_[capacity_ - 1] >>=
255  (uint64_t(capacity_ - 1) > target_word) ? 1 : 0;
256  size_--;
257  return x;
258  }

273  int set(const uint64_t i, const bool x) {
274  uint64_t idx = i;
275  if constexpr (buffer_size != 0) {
276  // If buffer exists, the index needs for the underlying structure
277  // needs to be modified. And if there exists an insertion to this
278  // location in the buffer, the insertion can simply be amended.
279  for (uint8_t j = 0; j < buffer_count_; j++) {
280  uint64_t b = buffer_index(buffer_[j]);
281  if (b < i) {
282  [[likely]] idx += buffer_is_insertion(buffer_[j]) ? -1 : 1;
283  } else if (b == i) {
284  if (buffer_is_insertion(buffer_[j])) {
285  if (buffer_value(buffer_[j]) != x) {
286  int change = x ? 1 : -1;
287  p_sum_ += change;
288  buffer_[j] ^= VALUE_MASK;
289  [[likely]] return change;
290  }
291  [[likely]] return 0;
292  }
293  idx++;
294  } else {
295  break;
296  }
297  }

323  uint64_t count = 0;

325  uint64_t idx = n;
326  if constexpr (buffer_size != 0) {

364  uint32_t select(const uint32_t x) const {
365  uint32_t pop = 0;
366  uint32_t pos = 0;
367  uint8_t current_buffer = 0;
368  int8_t a_pos_offset = 0;
369 
370  // Step one 64-bit word at a time considering the buffer until pop >= x
371  for (uint64_t j = 0; j < capacity_; j++) {
372  pop += __builtin_popcountll(data_[j]);
373  pos += 64;
374  if constexpr (buffer_size != 0) {
375  for (uint8_t b = current_buffer; b < buffer_count_; b++) {
376  uint64_t b_index = buffer_index(buffer_[b]);
377  if (b_index < pos) {
378  if (buffer_is_insertion(buffer_[b])) {
379  pop += buffer_value(buffer_[b]);
380  pos++;
381  a_pos_offset--;
382  } else {
383  pop -=
384  (data_[(b_index + a_pos_offset) / WORD_BITS] >>
385  ((b_index + a_pos_offset) % WORD_BITS)) &
386  MASK;
387  pos--;
388  a_pos_offset++;
389  }
390  [[unlikely]] current_buffer++;
391  } else {
392  break;
393  }
394  }
395  }
396  if (pop >= x) [[unlikely]]
397  break;
398  }
399 
400  // Make sure we have not overshot the logical end of the structure.
401  pos = size_ < pos ? size_ : pos;
402 
403  // Decrement one bit at a time until we can't anymore without going
404  // under x.
405  pos--;
406  while (pop >= x && pos < capacity_ * 64) {
407  pop -= at(pos);
408  pos--;
409  }
410  return ++pos;
411  }

482  void clear_first(uint64_t elems) {
483  // Important to ensure that all trailing data gets zeroed out to not
484  // cause undefined behaviour for select or copy operations.
485  uint64_t ones = rank(elems);
486  uint64_t words = elems / WORD_BITS;
487 
488  if (elems % WORD_BITS == 0) {
489  // If the data to remove happens to align with 64-bit words the
490  // removal can be done by simply shuffling elements. This could be
491  // memmove instead but testing indicates that it's not actually
492  // faster in this case.
493  for (uint64_t i = 0; i < capacity_ - words; i++) {
494  data_[i] = data_[i + words];
495  }
496  for (uint64_t i = capacity_ - words; i < capacity_; i++) {
497  data_[i] = 0;
498  }
499  } else {
500  // clear elem bits at start of data
501  for (uint64_t i = 0; i < words; i++) {
502  data_[i] = 0;
503  }
504  uint64_t tail = elems % WORD_BITS;
505  uint64_t tail_mask = (MASK << tail) - 1;
506  data_[words] &= ~tail_mask;
507 
508  // Copy data backwars by elem bits
509  uint64_t words_to_shuffle = capacity_ - words - 1;
510  for (uint64_t i = 0; i < words_to_shuffle; i++) {
511  data_[i] = data_[words + i] >> tail;
512  data_[i] |= (data_[words + i + 1]) << (WORD_BITS - tail);
513  }
514  data_[capacity_ - words - 1] = data_[capacity_ - 1] >> tail;
515  for (uint64_t i = capacity_ - words; i < capacity_; i++) {
516  data_[i] = 0;
517  }
518  [[unlikely]] (void(0));
519  }
520  size_ -= elems;
521  p_sum_ -= ones;
522  }

537  void transfer_append(leaf* other, uint64_t elems) {
538  commit();
539  other->commit();
540 
541  uint64_t* o_data = other->data();
542  uint64_t split_point = size_ % WORD_BITS;
543  uint64_t target_word = size_ / WORD_BITS;
544  uint64_t copy_words = elems / WORD_BITS;
545  uint64_t overhang = elems % WORD_BITS;
546  // Branching depending on word alignment of source and target.
547  if (split_point == 0) {
548  // Copy target is word aligned
549  for (size_t i = 0; i < copy_words; i++) {
550  data_[target_word++] = o_data[i];
551  p_sum_ += __builtin_popcountll(o_data[i]);
552  }
553  if (overhang != 0) {
554  // Copy source is not word aligned
555  data_[target_word] =
556  o_data[copy_words] & ((MASK << overhang) - 1);
557  [[unlikely]] p_sum_ += __builtin_popcountll(data_[target_word]);
558  }
559  } else {
560  // Copy target is not word aligned
561  for (size_t i = 0; i < copy_words; i++) {
562  data_[target_word++] |= o_data[i] << split_point;
563  data_[target_word] |= o_data[i] >> (WORD_BITS - split_point);
564  p_sum_ += __builtin_popcountll(o_data[i]);
565  }
566  if (elems % WORD_BITS != 0) {
567  // Copy source is not word aligned.
568  uint64_t to_write =
569  o_data[copy_words] & ((MASK << overhang) - 1);
570  p_sum_ += __builtin_popcountll(to_write);
571  data_[target_word++] |= to_write << split_point;
572  if (target_word < capacity_) {
573  data_[target_word] |= to_write >> (WORD_BITS - split_point);
574  }
575  [[likely]] ((void)0);
576  }
577  }
578  size_ += elems;
579  other->clear_first(elems);
580  }

620  void transfer_prepend(leaf* other, uint64_t elems) {
621  commit();
622  other->commit();
623  uint64_t* o_data = other->data();
624  uint64_t words = elems / 64;
625  // Make space for new data
626  for (uint64_t i = capacity_ - 1; i >= words; i--) {
627  data_[i] = data_[i - words];
628  if (i == 0) break;
629  }
630  for (uint64_t i = 0; i < words; i++) {
631  data_[i] = 0;
632  }
633  uint64_t overflow = elems % 64;
634  if (overflow > 0) {
635  for (uint64_t i = capacity_ - 1; i > words; i--) {
636  data_[i] <<= overflow;
637  data_[i] |= data_[i - 1] >> (64 - overflow);
638  }
639  data_[words] <<= overflow;
640  [[likely]] (void(0));
641  }
642  // Copy over data from sibling
643  uint64_t source_word = other->size();
644  uint64_t source_offset = source_word % 64;
645  source_word /= 64;
646  // Somewhat complicated case handling based on target and source word
647  // alignment. Additional complications compared to transfer append is
648  // due to the both the start point and end point for the copy source not
649  // being word aligned.
650  if (source_offset == 0) {
651  // Source is word aligned.
652  if (overflow == 0) {
653  // Target is word aligned.
654  for (uint64_t i = words - 1; i < words; i--) {
655  data_[i] = o_data[--source_word];
656  p_sum_ += __builtin_popcountll(data_[i]);
657  }
658  } else {
659  // Target is not word aligned.
660  source_word--;
661  for (uint64_t i = words; i > 0; i--) {
662  p_sum_ += __builtin_popcountll(o_data[source_word]);
663  data_[i] |= o_data[source_word] >> (64 - overflow);
664  data_[i - 1] |= o_data[source_word--] << overflow;
665  }
666  uint64_t w = o_data[source_word] >> (64 - overflow);
667  p_sum_ += __builtin_popcountll(w);
668  [[likely]] data_[0] |= w;
669  }
670  [[unlikely]] (void(0));
671  } else {
672  // Source is not word aligned
673  if (overflow == 0) {
674  // Target is word aligned
675  for (uint64_t i = words - 1; i < words; i--) {
676  data_[i] = o_data[source_word] << (64 - source_offset);
677  data_[i] |= o_data[--source_word] >> source_offset;
678  p_sum_ += __builtin_popcountll(data_[i]);
679  }
680  } else {
681  // Target is not word aligned
682  uint64_t w;
683  for (uint64_t i = words; i > 0; i--) {
684  w = o_data[source_word] << (64 - source_offset);
685  w |= o_data[--source_word] >> source_offset;
686  p_sum_ += __builtin_popcountll(w);
687  data_[i] |= w >> (64 - overflow);
688  data_[i - 1] |= w << overflow;
689  }
690  w = o_data[source_word] << (64 - source_offset);
691  w |= o_data[--source_word] >> source_offset;
692  w >>= 64 - overflow;
693  p_sum_ += __builtin_popcountll(w);
694  [[likely]] data_[0] |= w;
695  }
696  }
697  size_ += elems;
698  other->clear_last(elems);
699  }

754  void commit() {
755  // Complicated bit manipulation but whacha gonna do. Hopefully won't
756  // need to debug this anymore.
757  if constexpr (buffer_size == 0) return;
758  if (buffer_count_ == 0) [[unlikely]]
759  return;
760 
761  uint64_t overflow = 0;
762  uint8_t overflow_length = 0;
763  uint8_t underflow_length = 0;
764  uint8_t current_index = 0;
765  uint32_t buf = buffer_[current_index];
766  uint64_t target_word = buffer_index(buf) / WORD_BITS;
767  uint64_t target_offset = buffer_index(buf) % WORD_BITS;
768 
769  uint64_t words = size_ / WORD_BITS;
770  words += size_ % WORD_BITS > 0 ? 1 : 0;
771  for (uint64_t current_word = 0; current_word < words; current_word++) {
772  uint64_t underflow =
773  current_word + 1 < capacity_ ? data_[current_word + 1] : 0;
774  if (overflow_length) {
775  [[likely]] underflow =
776  (underflow << overflow_length) |
777  (data_[current_word] >> (64 - overflow_length));
778  }
779 
780  uint64_t new_overflow = 0;
781  // If buffers need to be commit to this word:
782  if (current_word == target_word && current_index < buffer_count_) {
783  uint64_t word =
784  underflow_length
785  ? (data_[current_word] >> underflow_length) |
786  (underflow << (64 - underflow_length))
787  : (data_[current_word] << overflow_length) | overflow;
788  underflow >>= underflow_length;
789  uint64_t new_word = 0;
790  uint8_t start_offset = 0;
791  // While there are buffers for this word
792  while (current_word == target_word) {
793  new_word |=
794  (word << start_offset) & ((MASK << target_offset) - 1);
795  word = (word >> (target_offset - start_offset)) |
796  (target_offset == 0 ? 0
797  : target_offset - start_offset == 0
798  ? 0
799  : (underflow
800  << (64 - (target_offset - start_offset))));
801  underflow >>= target_offset - start_offset;
802  if (buffer_is_insertion(buf)) {
803  if (buffer_value(buf)) {
804  new_word |= MASK << target_offset;
805  }
806  start_offset = target_offset + 1;
807  if (underflow_length) [[unlikely]]
808  underflow_length--;
809  else
810  overflow_length++;
811  } else {
812  word >>= 1;
813  word |= underflow << 63;
814  underflow >>= 1;
815  if (overflow_length)
816  overflow_length--;
817  else [[likely]]
818  underflow_length++;
819  start_offset = target_offset;
820  }
821  current_index++;
822  if (current_index >= buffer_count_) [[unlikely]]
823  break;
824  buf = buffer_[current_index];
825  target_word = buffer_index(buf) / WORD_BITS;
826  [[unlikely]] target_offset = buffer_index(buf) % WORD_BITS;
827  }
828  new_word |=
829  start_offset < 64 ? (word << start_offset) : uint64_t(0);
830  new_overflow = overflow_length ? data_[current_word] >>
831  (64 - overflow_length)
832  : 0;
833  [[unlikely]] data_[current_word] = new_word;
834  } else {
835  if (underflow_length) {
836  data_[current_word] =
837  (data_[current_word] >> underflow_length) |
838  (underflow << (64 - underflow_length));
839  } else if (overflow_length) {
840  new_overflow =
841  data_[current_word] >> (64 - overflow_length);
842  [[likely]] data_[current_word] =
843  (data_[current_word] << overflow_length) | overflow;
844  } else {
845  overflow = 0;
846  }
847  }
848  overflow = new_overflow;
849  }
850  if (capacity_ > words) [[likely]]
851  data_[words] = 0;
852  buffer_count_ = 0;
853  }
