682  constexpr dtype num_bits = sizeof(dtype) * 8;
683  constexpr dtype lines = CACHE_LINE / sizeof(dtype);
684  for (dtype i = 0; i < branches; i += lines) {
685  __builtin_prefetch(child_sizes_ + i);

688  if constexpr (branches == 128) {
689  idx = (uint8_t(1) << 6) - 1;
690  idx ^=
691  (dtype((child_sizes_[idx] - q) & SIGN_BIT) >> (num_bits - 7)) |
692  (uint8_t(1) << 5);
693  idx ^=
694  (dtype((child_sizes_[idx] - q) & SIGN_BIT) >> (num_bits - 6)) |
695  (uint8_t(1) << 4);
696  idx ^=
697  (dtype((child_sizes_[idx] - q) & SIGN_BIT) >> (num_bits - 5)) |

840  template <class allocator>
841  void rebalance_leaf(uint8_t index, leaf_type* leaf, allocator* alloc) {
842  // Number of leaves that can fit in the "left" sibling (with potential
843  // reallocation).
844  uint32_t l_cap = 0;
845  if (index > 0) {
846  l_cap = child_sizes_[index - 1];
847  l_cap -= index > 1 ? child_sizes_[index - 2] : 0;
848  [[likely]] l_cap = leaf_size - l_cap;
849  }
850  // Number of leaves that can fir in the "right" sibling (with potential
851  // reallocation).
852  uint32_t r_cap = 0;
853  if (index < child_count() - 1) {
854  r_cap = child_sizes_[index + 1];
855  r_cap -= child_sizes_[index];
856  [[likely]] r_cap = leaf_size - r_cap;
857  }
858  if (l_cap < 2 * leaf_size / 9 && r_cap < 2 * leaf_size / 9) {
859  // Rebalancing without creating a new leaf is impossible
860  // (impracitcal).
861  leaf_type* a_child;
862  leaf_type* b_child;
863  leaf_type* new_child;
864  if (index == 0) {
865  // If the full leaf is the first child, a new leaf is created
866  // between indexes 0 and 1.
867  dtype n_elem = child_sizes_[1] / 3;
868  dtype n_cap = (n_elem + 128) / 64;
869  n_cap += n_cap % 2;
870  n_cap = n_cap * 64 > leaf_size ? leaf_size / 64 : n_cap;
871  a_child = reinterpret_cast<leaf_type*>(children_[0]);
872  new_child = alloc->template allocate_leaf<leaf_type>(n_cap);
873  b_child = reinterpret_cast<leaf_type*>(children_[1]);
874  new_child->transfer_append(b_child, b_child->size() - n_elem);
875  new_child->transfer_prepend(a_child, a_child->size() - n_elem);
876  [[unlikely]] index++;
877  } else {
878  // If the full leaf is not the first child, a new leaf is
879  // created to the left of the full leaf.
880  a_child = reinterpret_cast<leaf_type*>(children_[index - 1]);
881  b_child = reinterpret_cast<leaf_type*>(children_[index]);
882  dtype n_elem = (a_child->size() + b_child->size()) / 3;
883  dtype n_cap = (n_elem + 128) / 64;
884  n_cap += n_cap % 2;
885  n_cap = n_cap * 64 > leaf_size ? leaf_size / 64 : n_cap;
886  new_child = alloc->template allocate_leaf<leaf_type>(n_cap);
887  new_child->transfer_append(b_child, b_child->size() - n_elem);
888  new_child->transfer_prepend(a_child, a_child->size() - n_elem);
889  }
890  // Update cumulative sizes and sums.
891  for (uint8_t i = child_count(); i > index; i--) {
892  child_sizes_[i] = child_sizes_[i - 1];
893  child_sums_[i] = child_sums_[i - 1];
894  children_[i] = children_[i - 1];
895  }
896  if (index == 1) {
897  child_sizes_[0] = a_child->size();
898  [[unlikely]] child_sums_[0] = a_child->p_sum();
899  } else {
900  child_sizes_[index - 1] =
901  child_sizes_[index - 2] + a_child->size();
902  child_sums_[index - 1] =
903  child_sums_[index - 2] + a_child->p_sum();
904  }
905  child_sizes_[index] = child_sizes_[index - 1] + new_child->size();
906  child_sums_[index] = child_sums_[index - 1] + new_child->p_sum();
907  children_[index] = new_child;
908  [[unlikely]] meta_data_++;
909  } else if (r_cap > l_cap) {
910  // Right sibling has more space than the left sibling. Move elements
911  // to right sibling
912  leaf_type* sibling =
913  reinterpret_cast<leaf_type*>(children_[index + 1]);
914  uint32_t n_size = sibling->size() + r_cap / 2;
915  if (sibling->capacity() * WORD_BITS < n_size) {
916  n_size = n_size / WORD_BITS + 1;
