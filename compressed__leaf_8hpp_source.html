13  // A 64-bit dtype implies that there are an even number of elements in the
14  // buffer. Since the buffer elements are 32 bits wide, storing 64-bit
15  // indexes requires 2 elements.
16  static_assert(sizeof(dtype) == sizeof(uint32_t) || buffer_size % 2 == 0);
17  // sizeof(dtype) = 8 => buffer_size <= 40, since each 64-bit element
18  // requires a total of 3 status bits and there are a total of 60 bits
19  // available.
20  static_assert(sizeof(dtype) == sizeof(uint32_t) || buffer_size <= 40);
21  // sizeof(dtype) = 4 => buffer_size <= 20, since each 32-bit element
22  // requires a total of 3 status bits and there are a total of 60 bits
23  // available.
24  static_assert(sizeof(dtype) == sizeof(uint64_t) || buffer_size <= 20);
25  // Only 31 and 63 bit universe sizes are supported at the internal node
26  // level so only the associated sizes are supported at the leaf level as
27  // well.

29  sizeof(dtype) == sizeof(uint32_t));
